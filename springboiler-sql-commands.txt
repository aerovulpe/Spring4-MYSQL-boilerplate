CREATE DATABASE springboilerplate;
USE springboilerplate;
CREATE TABLE accounts (id int(10) NOT NULL AUTO_INCREMENT PRIMARY KEY, naturalId varchar(100) NOT NULL UNIQUE, password varchar(60), firstName varchar(40) NOT NULL, lastName varchar(50) NOT NULL, email varchar(50) NOT NULL, gender varchar(15), locale varchar(40), pictureUrl varchar(100), location varchar(100));
CREATE TABLE roles (accountId int(10) NOT NULL, role varchar(25) NOT NULL);
CREATE TABLE permissions (accountId int(10) NOT NULL, permission varchar(50) NOT NULL);
CREATE USER 'springboilerplateserver'@'localhost' IDENTIFIED BY 'password';
GRANT ALL PRIVILEGES ON * . * TO 'springboilerplateserver'@'localhost';
FLUSH PRIVILEGES;
